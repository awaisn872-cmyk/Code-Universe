  • 5. Loops
    • // for loop
for(int i=0;i<5;i++){
    printf("%d\n", i);
}

// while loop
int i=0;
while(i<5){
    printf("%d\n", i);
    i++;
}

// do-while loop
int j=0;
do{
    printf("%d\n", j);
    j++;
}while(j<5);
