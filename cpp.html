  • 4. Conditions (if, else, switch)
    • int age = 18;

if(age >= 18){
    cout << "Adult";
} else {
    cout << "Minor";
}

switch(age){
    case 18:
        cout << "Exactly 18";
        break;
    default:
        cout << "Other";
}
  • 5. Loops (for, while, do-while)
    • // for loop
for(int i=0;i<5;i++){
    cout << i;
}

// while loop
int i=0;
while(i<5){
    cout << i;
    i++;
}

// do-while loop
int j=0;
do{
    cout << j;
    j++;
}while(j<5);

  • 9. Object-Oriented Programming (OOP)
    • class Person{
public:
    string name;
    int age;

    void greet(){
        cout << "Hello " << name;
    }
};

int main(){
    Person p;
    p.name = "Alice";
    p.age = 25;
    p.greet();
    return 0;
}
